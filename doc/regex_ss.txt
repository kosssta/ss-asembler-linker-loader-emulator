#include <iostream>
#include <string>
#include <regex>

using namespace std;

int main()
{
    smatch sm;
    string str = " labela : mov     rax  , rbx ";
    regex_match(str, sm, regex("\\s*(?:([A-Za-z][A-Za-z0-9_]*)\\s*:\\s*){0,1}(?:([A-Za-z]+)\\s+([A-Za-z0-9_]+)\\s*(?:,\\s*([A-Za-z0-9_]+)\\s*)*){0,1}"));
    
    for(unsigned i = 1; i < sm.size(); ++i) {
        cout << '[' << sm[i] << ']' << endl;
    }
    return 0;
}
